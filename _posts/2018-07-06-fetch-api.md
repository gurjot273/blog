function use_ajax(num)
{

       credentials: 'same-origin',
       headers: {
       				    'Content-Type': 'application/json',
    			      },
       body: json_upload })
       .then(function(response) {
              if(response.ok)
              {
                return (response.json());
              }})
       .then(obj => { var item=obj[0];
                      alert('Response '+ item + ' received');
                    });
}
